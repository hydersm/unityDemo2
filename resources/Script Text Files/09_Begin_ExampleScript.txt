using UnityEngine;
using System.Collections;

public class ExampleScript_B : MonoBehaviour 
{
	public int weapon = 0;
	public string weaponName;

	// Use this for initialization
	void Start () 
	{

	}
	
	// Update is called once per frame
	void Update () 
	{
		if(Input.GetKeyUp ("space"))
		{
			WeaponSearch();
			DisplayMessage();
		}
	}

	void WeaponSearch()
	{
		weapon = Random.Range (1, 20);
		
		switch(weapon)
		{
		case 1: 
			weaponName = "sword";
			break;
		case 2: 
			weaponName = "axe";
			break;
		case 3:
			weaponName = "dagger";
			break;
		case 4:
			weaponName = "bow";
			break;
		default:
			print ("You didn't find anything!");
			break;
		}
	}

	void DisplayMessage()
	{
		if(weapon < 5)
		{
			print ("You found the " + weaponName + "!");
		}
	}
}
